CREATE VIEW q1iii(birthyear, avgheight, count)
AS
  SELECT birthyear, AVG(height), COUNT(birthyear) 
        FROM people 
        GROUP BY birthyear 
        ORDER BY birthyear

;

-- Question 1iv
CREATE VIEW q1iv(birthyear, avgheight, count)
AS
  SELECT birthyear, AVG(height), COUNT(birthyear)
        FROM people
        GROUP BY birthyear
        HAVING COUNT(height) > 70 ORDER BY birthyear
;
